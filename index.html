function fetchFromJANAPI(code, callback) { const url = `https://api.janapi.jp/api/jan/${code}?apikey=${API_KEY}`; fetch(url) .then(res => res.json()) .then(data => { console.log("JANAPI 応答:", data); // 🔍 ここでログ出力 // 日本語名を探す const jp = data.name_jp || data.name_japanese || data.product_name_ja; const en = data.name; const name = jp || en || null; const price = data.price || null; if (name) callback({ name, price }); else callback(null); }) .catch(() => callback(null)); }
